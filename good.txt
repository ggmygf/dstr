    <script>
        const canvas = document.getElementById('chart');
        const ctx = canvas.getContext('2d');
        
        let charts = JSON.parse(localStorage.getItem('saved_charts')) || [{ id: 'BTCUSDT_15', symbol: 'BTCUSDT', interval: '15', title: 'BTC 15m', suffix: '' }];
        let activeIdx = 0;
         
        let candles = [];
        let trendlines = [];
        let scaleX = 0.5, scaleY = 10, offsetX = 0, offsetY = 0;
        const spacing = 2;

        let hasInitializedZoom = false;
        let isDrawingMode = false;
        let activePoints = [];
        let currentDotIndex = -1;
        let targetType = 'high';
        let drawingStep = 1;
        let selectedLineIndex = -1;
        let lastCreateAttempt = null;

        let isDragging = false, isLineDragging = false, isPinching = false, isSwipeScaling = false;
        let startDragX = 0, startDragY = 0, startOffsetX = 0, startOffsetY = 0, startPinchDist = 0, startScaleX = 0.5;
        let swipeStartY = 0, swipeStartScaleY = 10;
        let dragStartPointer = null, dragOriginalLine = null;
        let lastTapTime = 0;

        function getActiveUnit() { return charts[activeIdx] || charts[0]; }

        function showToast(msg, type = 'good') {
            const t = document.getElementById('toast');
            t.textContent = msg;
            t.className = type === 'good' ? 'toast-good' : 'toast-bad';
            t.style.display = 'block';
            setTimeout(() => t.style.display = 'none', 2500);
        }

        async function fetchSmart(isManual = false) {
            const unit = getActiveUnit();
            if (!unit) return;
            
            try {
                const response = await fetch(`https://api.bybit.com/v5/market/kline?category=linear&symbol=${unit.symbol}&interval=${unit.interval}&limit=200`);
                const data = await response.json();
                
                if (data.result && data.result.list) {
                    const fetched = data.result.list.map(d => ({
                        timestamp: parseInt(d[0]), open: parseFloat(d[1]), high: parseFloat(d[2]), low: parseFloat(d[3]), close: parseFloat(d[4])
                    }));
                    
                    let newCount = 0;
                    let kMap = new Map();
                    candles.forEach(c => kMap.set(c.timestamp, c));
                    
                    fetched.forEach(c => {
                        if (!kMap.has(c.timestamp)) newCount++;
                        kMap.set(c.timestamp, c);
                    });
                    
                    candles = Array.from(kMap.values()).sort((a, b) => a.timestamp - b.timestamp);
                    localStorage.setItem(`k_${unit.id}`, JSON.stringify(candles));
                    
                    if (isManual) {
                        if (newCount > 0) showToast(`Good: Updated ${newCount} new candles.`, 'good');
                        else showToast(`Up to date. No new candles.`, 'good');
                    }
                    
                    if (!hasInitializedZoom && candles.length > 0) {
                        hasInitializedZoom = true;
                        scaleX = 0.5; // Setup minimum configuration width defaults
                        saveViewState();
                    }
                    draw();
                }
            } catch (e) {
                if (isManual) showToast("Bad: Fetch failed.", "bad");
            }
        }

        function switchChart(idx) {
            if (charts.length === 0) return;
            saveViewState(); 
            
            activeIdx = (idx + charts.length) % charts.length;
            const unit = getActiveUnit();
            
            const baseAsset = unit.symbol.replace("USDT", "");
            const tfLabel = unit.interval === '15' ? '15m' : unit.interval === '60' ? 'H1' : unit.interval === '240' ? 'H4' : unit.interval === 'D' ? 'D' : 'W';
            const cleanSpec = `${baseAsset} ${tfLabel} `;
            const suffixStr = unit.suffix || "";
            document.getElementById('window-title').textContent = cleanSpec + suffixStr;
            
            candles = JSON.parse(localStorage.getItem(`k_${unit.id}`) || '[]');
            trendlines = JSON.parse(localStorage.getItem(`lines_${unit.id}`) || '[]');
            
            const viewState = JSON.parse(localStorage.getItem(`view_${unit.id}`) || '{"scaleX":0.5,"scaleY":10,"offsetX":0,"offsetY":0,"hasInitializedZoom":false}');
            scaleX = Math.max(0.5, viewState.scaleX); 
            scaleY = viewState.scaleY; 
            offsetX = viewState.offsetX;
            offsetY = viewState.offsetY || 0;
            hasInitializedZoom = viewState.hasInitializedZoom;

            isDrawingMode = false;
            selectedLineIndex = -1; activePoints = [];
            document.getElementById('btn-draw').classList.remove('active');
            document.getElementById('draw-sub-inputs').style.display = 'none';
            updateDeleteButton();
            
            draw();
            if (candles.length === 0) fetchSmart(false);
        }

        function saveViewState() {
            if (!charts[activeIdx]) return;
            localStorage.setItem(`view_${charts[activeIdx].id}`, JSON.stringify({scaleX, scaleY, offsetX, offsetY, hasInitializedZoom}));
            localStorage.setItem('saved_charts', JSON.stringify(charts));
        }

        function saveLines() {
            if (!charts[activeIdx]) return;
            localStorage.setItem(`lines_${charts[activeIdx].id}`, JSON.stringify(trendlines));
        }

        function getXByTimestamp(ts) {
            const w = canvas.width / window.devicePixelRatio;
            const startX = w - 80 + offsetX;
            const idx = candles.findIndex(c => c.timestamp === ts);
            if (idx === -1) return -9999;
            const idxFromEnd = candles.length - 1 - idx;
            return startX - (idxFromEnd * (scaleX + spacing));
        }

        function getY(p) {
            const h = canvas.height / window.devicePixelRatio;
            let minP = Infinity, maxP = -Infinity;
            candles.forEach(c => { if (c.low < minP) minP = c.low; if (c.high > maxP) maxP = c.high; });
            const priceRange = maxP - minP;
            const centerY = h / 2;
            const centerPrice = minP + (priceRange / 2);
            if (priceRange === 0) return centerY + offsetY; 
            return centerY + (centerPrice - p) * (scaleY * (h / priceRange / 15)) + offsetY;
        }

        function resize() {
            const rect = document.getElementById('main-area').getBoundingClientRect();
            canvas.width = rect.width * window.devicePixelRatio;
            canvas.height = rect.height * window.devicePixelRatio;
            ctx.scale(window.devicePixelRatio, window.devicePixelRatio);
            draw();
        }

        function draw() {
            const w = canvas.width / window.devicePixelRatio;
            const h = canvas.height / window.devicePixelRatio;
            ctx.clearRect(0, 0, w, h);
            
            if (candles.length === 0) return;
            // Strict Canvas Edge Containment Matrix
            let minP = Infinity, maxP = -Infinity;
            candles.forEach(c => { if (c.low < minP) minP = c.low; if (c.high > maxP) maxP = c.high; });
            const yMaxPrice = getY(maxP);
            const yMinPrice = getY(minP);
            if (yMaxPrice < 0) { offsetY -= yMaxPrice; }
            else if (yMinPrice > h) { offsetY -= (yMinPrice - h); }
            const startX = w - 80 + offsetX; 

            // Candles Architecture Implementation
            candles.forEach((c, i) => {
                const idxFromEnd = candles.length - 1 - i;
                const x = startX - (idxFromEnd * (scaleX + spacing));
                if (x < -scaleX) return;

                const yOpen = getY(c.open), yClose = getY(c.close);
                const yHigh = getY(c.high), yLow = getY(c.low);
                const isBull = c.close >= c.open;
                
                // Color Specification: Up in orange, down in gray
                const color = isBull ? '#ff9500' : '#8e8e93';
                
                ctx.strokeStyle = color; ctx.fillStyle = color; ctx.lineWidth = 1.5;
                ctx.beginPath(); ctx.moveTo(x + scaleX / 2, yHigh); ctx.lineTo(x + scaleX / 2, yLow); ctx.stroke();
                ctx.fillRect(x, Math.min(yOpen, yClose), scaleX, Math.max(Math.abs(yClose - yOpen), 1));
            });

            // Trendlines
            trendlines.forEach((line, index) => {
                const x1 = getXByTimestamp(line.p1.idxFromEnd) + scaleX / 2, y1 = getY(line.p1.price);
                const x2 = getXByTimestamp(line.p2.idxFromEnd) + scaleX / 2, y2 = getY(line.p2.price);
                
                ctx.strokeStyle = (index === selectedLineIndex) ? '#58a6ff' : '#8b949e';
                ctx.lineWidth = (index === selectedLineIndex) ? 2.5 : 1.5;

                ctx.beginPath(); ctx.moveTo(x1, y1);
                const dx = x2 - x1, dy = y2 - y1, len = Math.hypot(dx, dy);
                if (len > 0.01) {
                    ctx.lineTo(x1 + (dx / len) * 10000, y1 + (dy / len) * 10000); ctx.stroke();
                }

                if (index === selectedLineIndex) {
                    ctx.fillStyle = '#1f6feb';
                }
            });

            activePoints.forEach(pt => {
                ctx.fillStyle = 'rgba(255, 255, 255, 0.75)'; ctx.beginPath();
                ctx.arc(getXByTimestamp(pt.idxFromEnd) + scaleX / 2, getY(pt.price), 3.5, 0, Math.PI * 2); ctx.fill();
            });

            // Hover Cursor
            if (isDrawingMode && currentDotIndex !== -1) {
                const tCan = candles.find(c => c.timestamp === currentDotIndex);
                if (tCan) {
                    ctx.fillStyle = '#ffffff';
                    ctx.strokeStyle = '#0d1117'; ctx.lineWidth = 1; ctx.beginPath();
                    ctx.arc(getXByTimestamp(currentDotIndex) + scaleX / 2, getY(targetType === 'high' ? tCan.high : tCan.low), 3, 0, Math.PI * 2);
                    ctx.fill(); ctx.stroke();
                }
            }
        }

        function updateHoverPoint(x, y) {
            const radius = 50;
            let bestDist = radius, bestCandle = null, bestType = null;
            const startX = (canvas.width / window.devicePixelRatio) - 80 + offsetX;
            for (let i = 0; i < candles.length; i++) {
                const cObj = candles[i];
                const cx = startX - ((candles.length - 1 - i) * (scaleX + spacing)) + scaleX / 2;
                if (Math.abs(x - cx) > radius) continue;
                
                const distHigh = Math.hypot(x - cx, y - getY(cObj.high));
                if (distHigh < bestDist) { bestDist = distHigh; bestCandle = cObj; bestType = 'high'; }

                const distLow = Math.hypot(x - cx, y - getY(cObj.low));
                if (distLow < bestDist) { bestDist = distLow; bestCandle = cObj; bestType = 'low'; }
            }

            if (bestCandle) { currentDotIndex = bestCandle.timestamp; targetType = bestType; return; }

            let minDist = Infinity;
            for (let i = 0; i < candles.length; i++) {
                const dist = Math.abs(x - (startX - ((candles.length - 1 - i) * (scaleX + spacing)) + scaleX / 2));
                if (dist < minDist) { minDist = dist; currentDotIndex = candles[i].timestamp; }
            }
            const tCan = candles.find(c => c.timestamp === currentDotIndex);
            if (tCan) targetType = Math.abs(y - getY(tCan.high)) < Math.abs(y - getY(tCan.low)) ? 'high' : 'low';
        }

        function onPointerDown(e) {
            if (isSwipeScaling) return;
            const touch = e.touches?.[0] ?? e.changedTouches?.[0];
            const cx = e.clientX !== undefined ? e.clientX : touch?.clientX ?? 0;
            const cy = (e.clientY !== undefined ? e.clientY : touch?.clientY ?? 0) - (document.getElementById('top-bar').getBoundingClientRect().height * (window.innerWidth > window.innerHeight ? 0 : 1));
            
            if (isDrawingMode) {
                if (cy < 100 && cx < 450) return;
                updateHoverPoint(cx, cy); draw(); return;
            }

            let clickedLineIdx = -1;
            trendlines.forEach((line, index) => {
                const x1 = getXByTimestamp(line.p1.idxFromEnd) + scaleX / 2, y1 = getY(line.p1.price);
                const x2 = getXByTimestamp(line.p2.idxFromEnd) + scaleX / 2, y2 = getY(line.p2.price);
                const A = cx - x1, B = cy - y1, C = x2 - x1, D = y2 - y1;
         
                const lenSq = C * C + D * D;
                const param = lenSq !== 0 ? (A * C + B * D) / lenSq : -1;
                const extX = param < 0 ? x1 : x1 + param * C, extY = param < 0 ? y1 : y1 + param * D;
         
                if (Math.hypot(cx - extX, cy - extY) < 25) clickedLineIdx = index;
            });

            if (clickedLineIdx !== -1) {
                selectedLineIndex = clickedLineIdx;
                isDragging = false; isLineDragging = true;
                dragStartPointer = { x: cx, y: cy }; dragOriginalLine = JSON.parse(JSON.stringify(trendlines[selectedLineIndex]));
            } else { 
                selectedLineIndex = -1;
                isDragging = true; 
                startDragX = cx - offsetX; 
                startDragY = cy - offsetY;
            }
            updateDeleteButton();
            draw();
        }

        function onPointerMove(e) {
            if (isSwipeScaling) return;
            const touch = e.touches?.[0] ?? e.changedTouches?.[0];
            const cx = e.clientX !== undefined ? e.clientX : touch?.clientX ?? 0;
            const cy = (e.clientY !== undefined ? e.clientY : touch?.clientY ?? 0) - (document.getElementById('top-bar').getBoundingClientRect().height * (window.innerWidth > window.innerHeight ? 0 : 1));
            
            if (isLineDragging && selectedLineIndex !== -1) {
                const candleShift = Math.round((cx - dragStartPointer.x) / (scaleX + spacing));
                let minP = Infinity, maxP = -Infinity;
                candles.forEach(c => { if (c.low < minP) minP = c.low; if (c.high > maxP) maxP = c.high; });
                const priceShift = -(cy - dragStartPointer.y) * ((maxP - minP) / (scaleY * (canvas.height/window.devicePixelRatio / 15)));
                
                const line = trendlines[selectedLineIndex];
                const p1Idx = candles.findIndex(c => c.timestamp === dragOriginalLine.p1.idxFromEnd);
                const p2Idx = candles.findIndex(c => c.timestamp === dragOriginalLine.p2.idxFromEnd);
                if (p1Idx !== -1) line.p1.idxFromEnd = candles[Math.max(0, Math.min(candles.length - 1, p1Idx + candleShift))]?.timestamp || line.p1.idxFromEnd;
                if (p2Idx !== -1) line.p2.idxFromEnd = candles[Math.max(0, Math.min(candles.length - 1, p2Idx + candleShift))]?.timestamp || line.p2.idxFromEnd;
                
                line.p1.price = dragOriginalLine.p1.price + priceShift;
                line.p2.price = dragOriginalLine.p2.price + priceShift;
                draw(); return;
            }

            if (!isDragging) return;
            offsetX = cx - startDragX;
            
            // Outer Space Prevention Pan Boundary Algorithm
            let candidateOffsetY = cy - startDragY;
            let originalOffsetY = offsetY;
            offsetY = candidateOffsetY;
            
            if (candles.length > 0) {
                let minP = Infinity, maxP = -Infinity;
                candles.forEach(c => { if (c.low < minP) minP = c.low; if (c.high > maxP) maxP = c.high; });
                
                const h = canvas.height / window.devicePixelRatio;
                const yMaxPrice = getY(maxP); // Upper price limit boundary
                const yMinPrice = getY(minP); // Lower price limit boundary
                
                if (yMaxPrice > h) { offsetY = h - (yMaxPrice - originalOffsetY); }
                if (yMinPrice < 0) { offsetY = 0 - (yMinPrice - originalOffsetY); }
            }
            draw();
        }

        function onPointerUp() { 
            isDragging = false;
            if (isLineDragging) saveLines(); 
            isLineDragging = false; 
            saveViewState(); 
        }

        canvas.addEventListener('mousedown', onPointerDown);
        window.addEventListener('mousemove', onPointerMove);
        window.addEventListener('mouseup', onPointerUp);

        canvas.addEventListener('touchstart', e => {
            if (e.touches.length === 2) {
                isPinching = true; isDragging = false; isLineDragging = false;
                startPinchDist = Math.hypot(e.touches[0].clientX - e.touches[1].clientX, e.touches[0].clientY - e.touches[1].clientY);
                startScaleX = scaleX; startOffsetX = offsetX; startOffsetY = offsetY;
            } else if (!isPinching) onPointerDown(e);
        });
        
        window.addEventListener('touchmove', e => {
            if (isPinching && e.touches.length === 2) {
                e.preventDefault(); 
                const dist = Math.hypot(e.touches[0].clientX - e.touches[1].clientX, e.touches[0].clientY - e.touches[1].clientY);
                const newScale = Math.max(0.5, Math.min(30, startScaleX * (dist / startPinchDist)));
             
                const cx = (e.touches[0].clientX + e.touches[1].clientX) / 2;
                const rm = window.innerWidth - 80;
                offsetX = cx - rm - ((cx - rm - startOffsetX) / (startScaleX + spacing)) * (newScale + spacing);
                scaleX = newScale; draw();
            } else if (!isPinching && e.touches.length === 1) onPointerMove(e);
        }, { passive: false });
        
        window.addEventListener('touchend', e => {
            if (e.touches.length < 2) isPinching = false;
            if (e.touches.length === 0) onPointerUp();
        });
        
        canvas.addEventListener('mousemove', e => {
            if (!isDrawingMode || isDragging || isPinching || isLineDragging) return;
            const rect = canvas.getBoundingClientRect();
            const x = (e.clientX ?? e.touches?.[0]?.clientX ?? 0) - rect.left;
            const y = (e.clientY ?? e.touches?.[0]?.clientY ?? 0) - rect.top;
            if (y < 100 && x < 450) return;
            updateHoverPoint(x, y); draw();
        });

            document.getElementById('btn-delete').style.display = hasSelection ? 'flex' : 'none';
            document.getElementById('btn-copy').style.display = hasSelection ? 'flex' : 'none';
            document.getElementById('btn-monitor').style.display = hasSelection ? 'flex' : 'none';
        }

            if (selectedLineIndex === -1) return;
            trendlines.splice(selectedLineIndex, 1); selectedLineIndex = -1;
            saveLines(); updateDeleteButton(); draw();
        });

            if (selectedLineIndex === -1) return;
            const o = trendlines[selectedLineIndex];
            trendlines.push({
                p1: { idxFromEnd: o.p1.idxFromEnd, price: o.p1.price + 100 }, p2: { idxFromEnd: o.p2.idxFromEnd, price: o.p2.price + 100 }
            });

        document.getElementById('btn-draw').addEventListener('click', () => {
            isDrawingMode = !isDrawingMode; selectedLineIndex = -1;
            activePoints = []; targetType = 'high'; drawingStep = 1;
            currentDotIndex = candles[Math.max(0, candles.length - 15)]?.timestamp || 0;
            document.getElementById('btn-draw').classList.toggle('active', isDrawingMode);
            document.getElementById('draw-sub-inputs').style.display = isDrawingMode ? 'flex' : 'none';
            document.getElementById('btn-simp').style.display = 'none';
            draw();
        });
        
        document.getElementById('btn-confirm').addEventListener('click', () => {
            if (!isDrawingMode || currentDotIndex === -1) return;
            const tc = candles.find(c => c.timestamp === currentDotIndex);
            if (!tc) return;
            const pVal = targetType === 'high' ? tc.high : tc.low;

            if (drawingStep === 1) {
                activePoints.push({ idxFromEnd: currentDotIndex, price: pVal });
                currentDotIndex = candles[Math.min(candles.length - 1, Math.max(0, candles.findIndex(c => c.timestamp === currentDotIndex) + 3))].timestamp;
                targetType = 'high'; drawingStep = 2;
                document.getElementById('btn-simp').style.display = 'inline-flex';
            } else if (drawingStep === 2) {
                activePoints.push({ idxFromEnd: currentDotIndex, price: pVal });
                trendlines.push({ p1: { ...activePoints[0] }, p2: { ...activePoints[1] } });
                activePoints = []; isDrawingMode = false; currentDotIndex = -1; drawingStep = 1;
                document.getElementById('btn-draw').classList.remove('active');
                document.getElementById('draw-sub-inputs').style.display = 'none';
                saveLines();
            }
            draw();
        });
        
        document.getElementById('btn-simp').addEventListener('click', () => {
            if (!isDrawingMode || drawingStep !== 2 || activePoints.length === 0) return;
            const p1 = activePoints[0];
            const p1Idx = candles.findIndex(c => c.timestamp === p1.idxFromEnd);
            if (p1Idx === -1) return;

            const targetIdx = Math.min(candles.length - 1, p1Idx + 3);
            const p2Candle = candles[targetIdx];
            
            trendlines.push({
                p1: { ...p1 },
                p2: { idxFromEnd: p2Candle.timestamp, price: p1.price }
            });
            
            activePoints = []; isDrawingMode = false; currentDotIndex = -1; drawingStep = 1;
            document.getElementById('btn-draw').classList.remove('active');
            document.getElementById('draw-sub-inputs').style.display = 'none';
            saveLines(); draw();
        });

        document.getElementById('btn-switch').addEventListener('click', () => { if (isDrawingMode) { targetType = targetType === 'high' ? 'low' : 'high'; draw(); } });
        document.getElementById('btn-left').addEventListener('click', () => {
            if (isDrawingMode && currentDotIndex !== -1) {
                const idx = candles.findIndex(c => c.timestamp === currentDotIndex);
                if (idx > 0) { currentDotIndex = candles[idx - 1].timestamp; draw(); }
            }
        });
        document.getElementById('btn-right').addEventListener('click', () => {
            if (isDrawingMode && currentDotIndex !== -1) {
                const idx = candles.findIndex(c => c.timestamp === currentDotIndex);
                if (idx !== -1 && idx < candles.length - 1) { currentDotIndex = candles[idx + 1].timestamp; draw(); }
            }
        });

            const p = document.getElementById('menu-pop'); p.style.display = p.style.display === 'block' ? 'none' : 'block';

        });

                document.getElementById('suffix-input').value = unit.suffix || "";
                document.getElementById('suffix-input').focus();

        });

            if (unit) {
                unit.suffix = document.getElementById('suffix-input').value.trim();

            }
            document.getElementById('suffix-modal').style.display = 'none';
        });

        // Seamless Drag-Vertical-Scale Handle Trackers
        touchZone.addEventListener('mousedown', (e) => {
            isSwipeScaling = true; swipeStartY = e.clientY; swipeStartScaleY = scaleY;
        });
        window.addEventListener('mousemove', (e) => {
            if (!isSwipeScaling) return;
            const deltaY = swipeStartY - e.clientY;
            scaleY = Math.max(0.2, swipeStartScaleY * Math.exp(deltaY * 0.006));
            saveViewState(); draw();
        });
        window.addEventListener('mouseup', () => { isSwipeScaling = false; });

        touchZone.addEventListener('touchstart', (e) => {
            if (e.touches.length === 1) {
                isSwipeScaling = true; swipeStartY = e.touches[0].clientY; swipeStartScaleY = scaleY;
                e.stopPropagation();
            }
        });
        window.addEventListener('touchmove', (e) => {
            if (!isSwipeScaling) return;
            const deltaY = swipeStartY - e.touches[0].clientY;
            scaleY = Math.max(0.2, swipeStartScaleY * Math.exp(deltaY * 0.006));
            saveViewState(); draw();
        });
        window.addEventListener('touchend', () => { isSwipeScaling = false; });

            const payload = {
                id: `${unit.id}_${line.p1.idxFromEnd}_${Date.now()}`,
                symbol: unit.symbol,

                p1: { timestamp: line.p1.idxFromEnd, price: line.p1.price },

            };

            try {
                const r = await fetch('http://18.162.56.220:2449/add', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify(payload)
                });
                if (r.ok) showToast("Task sent to monitor server.", "good");
                else showToast("Server refused monitor sync payload.", "bad");
            } catch(err) {
                showToast("Network pipeline context failure.", "bad");
            }
