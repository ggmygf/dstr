        <div class="controls">
            <button class="btn" id="w-plus" title="Stretch Width">W+</button>
            <button class="btn" id="w-minus" title="Shrink Width">W-</button>

            <button class="btn" id="h-minus" title="Shrink Height">H-</button>

    <script>
        const canvas = document.getElementById('chart');
        const ctx = canvas.getContext('2d');
        
        let candles = [];
        let scaleX = 8;
        let scaleY = 10;
        const spacing = 2; 
        let hasInitializedZoom = false;
        let isPointLocked = false;

        let offsetX = 0;
        let isDragging = false;
        let startDragX = 0;
        let isLineDragging = false;

        let isPinching = false;
        let startPinchDist = 0;
        let startScaleX = 8;
        let pinchStartCenterX = 0;
        let pinchStartCandleIdx = 0;
        let dragStartPointer = null;
        let dragOriginalLine = null;

        let trendlines = [];
        let activePoints = [];
        let isDrawingMode = false;
        let currentDotIndex = -1; // Timestamp of currently selected candle
        let targetType = 'high';
        let drawingStep = 1;
        let startOffsetX = 0;
        let selectedLineIndex = -1;

        // Global coordinate getters
        function getXByTimestamp(ts) {
            const w = window.innerWidth;
            const startX = w - 80 + offsetX;
            const idx = candles.findIndex(c => c.timestamp === ts);
            if (idx === -1) return -9999;
            const idxFromEnd = candles.length - 1 - idx;
            return startX - (idxFromEnd * (scaleX + spacing));
        }

        function getY(p) {
            const h = window.innerHeight;
            let minP = Infinity, maxP = -Infinity;
            candles.forEach(c => {
                if (c.low < minP) minP = c.low;
                if (c.high > maxP) maxP = c.high;
            });
            const priceRange = maxP - minP;
            const centerY = h / 2;
            const centerPrice = minP + (priceRange / 2);
            if (priceRange === 0) return centerY; 
            return centerY + (centerPrice - p) * (scaleY * (h / priceRange / 15));
        }

        function resize() {
            canvas.width = window.innerWidth * window.devicePixelRatio;
            canvas.height = window.innerHeight * window.devicePixelRatio;
            canvas.style.width = window.innerWidth + 'px';
            canvas.style.height = window.innerHeight + 'px';
            ctx.scale(window.devicePixelRatio, window.devicePixelRatio);
            draw();
        }

        async function fetchData() {
            try {
                const response = await fetch('https://api.bybit.com/v5/market/kline?category=linear&symbol=BTCUSDT&interval=15&limit=100');
                const data = await response.json();

                if (data.result && data.result.list) {
                    candles = data.result.list.map(d => ({
                        timestamp: parseInt(d[0]),
                        open: parseFloat(d[1]),
                        high: parseFloat(d[2]),
                        low: parseFloat(d[3]),
                        close: parseFloat(d[4])
                    })).reverse();
                    draw();

                    if (!hasInitializedZoom) {
                        hasInitializedZoom = true;
                        const hMinusBtn = document.getElementById('h-minus');
                        const wMinusBt = document.getElementById('w-minus');

                        for (let i = 0; i < 5; i++) {
                            hMinusBtn.click();
                            wMinusBt.click();
                        }
                    }
                }
            } catch (e) { console.error("Failed fetching data", e); }
        }

        // Smart Finger Snapping logic
        function updateHoverPoint(x, y) {
            const radius = 50; // Finger radius tolerance
            let bestDist = radius;
            let bestCandle = null;
            let bestType = null;
            
            const w = window.innerWidth;
            const startX = w - 80 + offsetX;

            // 1. Search inside the 50px radial bubble
            for (let i = 0; i < candles.length; i++) {
                const cObj = candles[i];
                const idxFromEnd = candles.length - 1 - i;
                const candleX = startX - (idxFromEnd * (scaleX + spacing)) + scaleX / 2;

                // Fast cull X
                if (Math.abs(x - candleX) > radius) continue;

                const yHigh = getY(cObj.high);
                const distHigh = Math.hypot(x - candleX, y - yHigh);
                if (distHigh < bestDist) {
                    bestDist = distHigh;
                    bestCandle = cObj;
                    bestType = 'high';
                }

                const yLow = getY(cObj.low);
                const distLow = Math.hypot(x - candleX, y - yLow);
                if (distLow < bestDist) {
                    bestDist = distLow;
                    bestCandle = cObj;
                    bestType = 'low';
                }
            }

            if (bestCandle) {
                currentDotIndex = bestCandle.timestamp;
                targetType = bestType;
                return;
            }

            // 2. Fallback: just snap to the closest X column if tap is outside 50px of any wick
            let minDist = Infinity;
            let selectedTimestamp = currentDotIndex;
            for (let i = 0; i < candles.length; i++) {
                const cObj = candles[i];
                const idxFromEnd = candles.length - 1 - i;
                const candleX = startX - (idxFromEnd * (scaleX + spacing)) + scaleX / 2;
                const dist = Math.abs(x - candleX);
                if (dist < minDist) {
                    minDist = dist;
                    selectedTimestamp = cObj.timestamp;
                }
            }
            currentDotIndex = selectedTimestamp;
            const targetCandle = candles.find(c => c.timestamp === currentDotIndex);
            if (targetCandle) {
                const distHigh = Math.abs(y - getY(targetCandle.high));
                const distLow = Math.abs(y - getY(targetCandle.low));
                targetType = distHigh < distLow ? 'high' : 'low';
            }
        }

        function draw() {
            const w = window.innerWidth;
            const h = window.innerHeight;
            ctx.clearRect(0, 0, w, h);
            
            if (candles.length === 0) return;
            
            const startX = w - 80 + offsetX; 

            // Draw Candles
            candles.forEach((c, i) => {
                const idxFromEnd = candles.length - 1 - i;
                const x = startX - (idxFromEnd * (scaleX + spacing));
                
                if (x < -scaleX) return;

                const yOpen = getY(c.open);
                const yClose = getY(c.close);
                const yHigh = getY(c.high);
                const yLow = getY(c.low);
                const isBull = c.close >= c.open;
                const color = isBull ? '#238636' : '#da3633';
        
                ctx.strokeStyle = color;
                ctx.fillStyle = color;
                ctx.lineWidth = 1.5;

                ctx.beginPath();
                ctx.moveTo(x + scaleX / 2, yHigh);
                ctx.lineTo(x + scaleX / 2, yLow);
                ctx.stroke();

                const bodyH = Math.abs(yClose - yOpen);
                ctx.fillRect(x, Math.min(yOpen, yClose), scaleX, Math.max(bodyH, 1));
            });

            // Draw Trendlines
            trendlines.forEach((line, index) => {
                const x1 = getXByTimestamp(line.p1.idxFromEnd) + scaleX / 2;
                const y1 = getY(line.p1.price);
                const x2 = getXByTimestamp(line.p2.idxFromEnd) + scaleX / 2;
                const y2 = getY(line.p2.price);
        const vP1 = line.visualP1 || line.p1;
        const vP2 = line.visualP2 || line.p2;
        const vx1 = getXByTimestamp(vP1.idxFromEnd) + scaleX / 2;
        const vy1 = getY(vP1.price);
        const vx2 = getXByTimestamp(vP2.idxFromEnd) + scaleX / 2;
        const vy2 = getY(vP2.price);
 
                if (index === selectedLineIndex) {
                    ctx.strokeStyle = '#58a6ff';  ctx.lineWidth = 2.5;
                } else { ctx.strokeStyle = '#8b949e'; ctx.lineWidth = 1.5;
                }

                ctx.beginPath(); ctx.moveTo(x1, y1);
                const dx = x2 - x1; const dy = y2 - y1;
        
                const len = Math.hypot(dx, dy);
                if (len > 0.01) {
                    const farX = x1 + (dx / len) * 10000;
                    const farY = y1 + (dy / len) * 10000;
                    ctx.lineTo(farX, farY); ctx.stroke();
                }

                // Draw Anchor Markers if Selected
                if (index === selectedLineIndex) {
                    ctx.fillStyle = '#1f6feb'; ctx.beginPath();
                    ctx.beginPath(); 
                    ctx.fillStyle = 'rgba(255, 255, 255, 0.4)';
        ctx.beginPath(); ctx.arc(vx1, vy1, 3, 0, Math.PI * 2); ctx.fill();
        ctx.beginPath(); ctx.arc(vx2, vy2, 3, 0, Math.PI * 2); ctx.fill();
                }
                
            });

            // Draw Active Points in Drawing Mode
            activePoints.forEach(pt => {
                const cx = getXByTimestamp(pt.idxFromEnd) + scaleX / 2;
                const cy = getY(pt.price);
                ctx.fillStyle = '#1f6feb';
                ctx.beginPath();
                ctx.arc(cx, cy, 2.5, 0, Math.PI * 2);
                ctx.fill();
            });

            // Draw Hover Point Cursor
            if (isDrawingMode && currentDotIndex !== -1) {
                const targetCandle = candles.find(c => c.timestamp === currentDotIndex);
                if (targetCandle) {
                    const cx = getXByTimestamp(currentDotIndex) + scaleX / 2;
                    const cy = getY(targetType === 'high' ? targetCandle.high : targetCandle.low);
                    ctx.fillStyle = '#ffffff';
                    ctx.strokeStyle = '#0d1117';
                    ctx.lineWidth = 1;
                    ctx.beginPath();
                    ctx.arc(cx, cy, 3, 0, Math.PI * 2);
                    ctx.fill();
                    ctx.stroke();
                }
            }
        }

        // Zoom & Pan modifiers
        document.getElementById('w-plus').addEventListener('click', () => { scaleX = Math.min(scaleX + 2, 30); draw(); });
        document.getElementById('w-minus').addEventListener('click', () => { scaleX = Math.max(scaleX - 2, 2); draw(); });
        document.getElementById('h-plus').addEventListener('click', () => { scaleY *= 1.2; draw(); });
        document.getElementById('h-minus').addEventListener('click', () => { scaleY /= 1.2; draw(); });

        window.addEventListener('resize', resize);
        resize(); fetchData(); setInterval(fetchData, 10000);

        function updateDeleteButton() {
            const deleteBtn = document.getElementById('btn-delete');
            const copyBtn = document.getElementById('btn-copy');

            if (selectedLineIndex !== -1) {
                deleteBtn.style.display = 'flex';
                copyBtn.style.display = 'flex';
            } else {
                deleteBtn.style.display = 'none';
                copyBtn.style.display = 'none';
            }
        }

        // Central Interaction Handler (Unified Touch/Mouse)
        function onPointerDown(e) {
            const touch = e.touches?.[0] ?? e.changedTouches?.[0];
            const clientX = e.clientX !== undefined ? e.clientX : touch?.clientX ?? 0;
            const clientY = e.clientY !== undefined ? e.clientY : touch?.clientY ?? 0;

            // Handle Draw Mode Click/Tap Snapping
            if (isDrawingMode) {
                if (clientY < 100 && clientX < 450) return; // Prevent snappping beneath UI buttons
                updateHoverPoint(clientX, clientY);
                isPointLocked = true;
                draw();
                return;
            }

            // Normal Mode: Detect Line Selection
            let clickedLineIdx = -1;
            const clickRadius = 25; // Greatly expanded for mobile hit-boxes

            trendlines.forEach((line, index) => {
                const x1 = getXByTimestamp(line.p1.idxFromEnd) + scaleX / 2;
                const y1 = getY(line.p1.price);
                const x2 = getXByTimestamp(line.p2.idxFromEnd) + scaleX / 2;
                const y2 = getY(line.p2.price);

                const A = clientX - x1;
                const B = clientY - y1;
                const C = x2 - x1;
                const D = y2 - y1;

                const dot = A * C + B * D;
                const lenSq = C * C + D * D;
                let param = -1;
                if (lenSq !== 0) param = dot / lenSq;

                let extX, extY;
                if (param < 0) {
                    extX = x1; extY = y1;
                } else {
                    extX = x1 + param * C;
                    extY = y1 + param * D;
                }

                const dist = Math.hypot(clientX - extX, clientY - extY);
                if (dist < clickRadius) {
                    clickedLineIdx = index;
                }
            });

            if (clickedLineIdx !== -1) {
                selectedLineIndex = clickedLineIdx;
                isDragging = false;
                isLineDragging = true;
                dragStartPointer = { x: clientX, y: clientY};
                dragOriginalLine = JSON.parse(JSON.stringify(trendlines[selectedLineIndex]));

                updateDeleteButton(); draw(); return;
            } else { 
                selectedLineIndex = -1; 
                updateDeleteButton(); draw(); 
            }

            // Background Panning
            if (selectedLineIndex === -1) {
                isDragging = true;
                startDragX = clientX - offsetX;
            }
        }

        function onPointerMove(e) {
            const touch = e.touches?.[0] ?? e.changedTouches?.[0];
            const clientX = e.clientX !== undefined ? e.clientX : touch?.clientX ?? 0;
            const clientY = e.clientY !== undefined ? e.clientY : touch?.clientY ?? 0;

            if (isLineDragging && selectedLineIndex !== -1) {
                const dx = clientX - dragStartPointer.x;
                const dy = clientY - dragStartPointer.y;
                const candleShift = Math.round(dx / (scaleX + spacing));
                
                let minP = Infinity, maxP = -Infinity;
                candles.forEach(c => {
                    if (c.low < minP) minP = c.low;
                    if (c.high > maxP) maxP = c.high;
                });
                const priceRange = maxP - minP;

                const pricePerPixel = priceRange / (scaleY * (window.innerHeight / 15));
                const priceShift = -dy * pricePerPixel;
                
                const line = trendlines[selectedLineIndex];
                const p1Idx = candles.findIndex(c => c.timestamp === dragOriginalLine.p1.idxFromEnd);
                const p2Idx = candles.findIndex(c => c.timestamp === dragOriginalLine.p2.idxFromEnd);
                
                if (p1Idx !== -1) {
                    const targetP1 = candles[Math.max(0, Math.min(candles.length - 1, p1Idx + candleShift))];
                    if (targetP1) line.p1.idxFromEnd = targetP1.timestamp;
                }
                if (p2Idx !== -1) {
                    const targetP2 = candles[Math.max(0, Math.min(candles.length - 1, p2Idx + candleShift))];
                    if (targetP2) line.p2.idxFromEnd = targetP2.timestamp;
                }
                
                line.p1.price = dragOriginalLine.p1.price + priceShift;
                line.p2.price = dragOriginalLine.p2.price + priceShift;
                draw(); return;
            }

            if (!isDragging) return; 
            offsetX = clientX - startDragX; 
            draw();
        }

        function onPointerUp() { isDragging = false; isLineDragging = false; }

        // Mouse events
        canvas.addEventListener('mousedown', onPointerDown);
        window.addEventListener('mousemove', onPointerMove);
        window.addEventListener('mouseup', onPointerUp);

        // Touch Pinch and Swipe Logic
        canvas.addEventListener('touchstart', function(e) {
            if (e.touches.length === 2) {
                isPinching = true;
                isDragging = false;
                isLineDragging = false;
                
                const dx = e.touches[0].clientX - e.touches[1].clientX;
                const dy = e.touches[0].clientY - e.touches[1].clientY;
                startPinchDist = Math.hypot(dx, dy);
                startScaleX = scaleX;
                startOffsetX = offsetX;
            } else if (e.touches.length === 1 && !isPinching) {
                onPointerDown(e);
            }
        });

        window.addEventListener('touchmove', function(e) {
            if (isPinching && e.touches.length === 2) {
                e.preventDefault(); 
                
                const dx = e.touches[0].clientX - e.touches[1].clientX;
                const dy = e.touches[0].clientY - e.touches[1].clientY;
                const currentPinchDist = Math.hypot(dx, dy);
                
                if (startPinchDist > 0) {
                    const factor = currentPinchDist / startPinchDist;
                    const newScaleX = Math.max(2, Math.min(30, startScaleX * factor));
                    
                    const currentCenterX = (e.touches[0].clientX + e.touches[1].clientX) / 2;
                    const rightMarginOffset = window.innerWidth - 80;
                
                    const totalCandleSizeOld = startScaleX + spacing;
                    const totalCandleSizeNew = newScaleX + spacing;
                    offsetX = currentCenterX - rightMarginOffset - 
                               ((currentCenterX - rightMarginOffset - startOffsetX) / totalCandleSizeOld) * totalCandleSizeNew;
                    scaleX = newScaleX;
                    draw();
                }
            } else if (!isPinching && e.touches.length === 1) {
                onPointerMove(e);
            }
        }, { passive: false });

        window.addEventListener('touchend', function(e) {
            if (e.touches.length < 2) { isPinching = false; }
            if (e.touches.length === 0) { onPointerUp();}
        });

        function updateActiveButtons() {
            document.getElementById('btn-draw').classList.toggle('active', isDrawingMode);
        }

        document.getElementById('btn-delete').addEventListener('click', () => {
            if (selectedLineIndex === -1) return;
            trendlines.splice(selectedLineIndex, 1);
            selectedLineIndex = -1;
            updateDeleteButton();
            draw();
        });

        document.getElementById('btn-copy').addEventListener('click', () => {
    if (selectedLineIndex === -1) return;
    const original = trendlines[selectedLineIndex];
    
    // Duplicate the line and create a deep copy of its unique visual anchors
    const copied = {
        p1: { idxFromEnd: original.p1.idxFromEnd, price: original.p1.price + 100 },
        p2: { idxFromEnd: original.p2.idxFromEnd, price: original.p2.price + 100 },
        visualP1: original.visualP1 ? { ...original.visualP1 } : { ...original.p1 },
        visualP2: original.visualP2 ? { ...original.visualP2 } : { ...original.p2 }
    };
    
    trendlines.push(copied);
    selectedLineIndex = trendlines.length - 1;
    updateDeleteButton();
    draw();
});

        document.getElementById('btn-draw').addEventListener('click', () => {
            isDrawingMode = true; selectedLineIndex = -1;
            activePoints = []; targetType = 'high'; drawingStep = 1;
            currentDotIndex = candles[Math.max(0, candles.length - 15)]?.timestamp || 0;
            updateActiveButtons(); draw();
        });

        document.getElementById('btn-confirm').addEventListener('click', () => {
            if (!isDrawingMode) return;
            if (currentDotIndex === -1) return;

            const targetCandle = candles.find(c => c.timestamp === currentDotIndex);
            if (!targetCandle) return;
            const priceVal = targetType === 'high' ? targetCandle.high : targetCandle.low;

            if (drawingStep === 1) {
                activePoints.push({ idxFromEnd: currentDotIndex, price: priceVal });
                
                const p1Idx = candles.findIndex(c => c.timestamp === currentDotIndex);
                const p2Idx = Math.min(candles.length - 1, Math.max(0, p1Idx + 3));
                
                currentDotIndex = candles[p2Idx].timestamp;
                targetType = 'high';
                drawingStep = 2;
            } else if (drawingStep === 2) {
                activePoints.push({ idxFromEnd: currentDotIndex, price: priceVal });
                trendlines.push({ 
            p1: { ...activePoints[0] }, 
            p2: { ...activePoints[1] },
            visualP1: { ...activePoints[0] },
            visualP2: { ...activePoints[1] }
        });
                activePoints = []; isDrawingMode = false; currentDotIndex = -1; drawingStep = 1;
                updateActiveButtons();
            }
            draw();
        });

        document.getElementById('btn-switch').addEventListener('click', () => {
            if (isDrawingMode) {
                targetType = targetType === 'high' ? 'low' : 'high';
                draw();
            }
        });

        // Fixed Left/Right Action logic using true Array Indices
        document.getElementById('btn-left').addEventListener('click', () => {
            if (isDrawingMode && currentDotIndex !== -1) {
                const idx = candles.findIndex(c => c.timestamp === currentDotIndex);
                if (idx > 0) { // Moving Left means querying Older data
                    currentDotIndex = candles[idx - 1].timestamp;
                    draw();
                }
            }
        });

        document.getElementById('btn-right').addEventListener('click', () => {
            if (isDrawingMode && currentDotIndex !== -1) {
                const idx = candles.findIndex(c => c.timestamp === currentDotIndex);
                if (idx !== -1 && idx < candles.length - 1) { // Moving Right means querying Newer data
                    currentDotIndex = candles[idx + 1].timestamp;
                    draw();
                }
            }
        });

        function handlePointerHover(e) {
            if (!isDrawingMode) return;
            if (isDragging || isPinching || isLineDragging) return;

            const rect = canvas.getBoundingClientRect();
            const x = (e.clientX ?? e.touches?.[0]?.clientX ?? 0) - rect.left;
            const y = (e.clientY ?? e.touches?.[0]?.clientY ?? 0) - rect.top;

            // Block UI elements
            if (y < 100 && x < 450) return; 

            updateHoverPoint(x, y);
            draw();
        }

        canvas.addEventListener('mousemove', handlePointerHover);
        canvas.addEventListener('touchmove', handlePointerHover, { passive: true });

    </script>
</body>
